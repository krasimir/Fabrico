Installation
Configuration\Fabrico
The configuration of Fabrico is stored in:
/config/config.json
The file should have the following format: 
{
    "access": {
        "user": "admin",
        "pass": "admin"
    },
    "paths": {
        "files": "/Fabrico",
        "http": "/fabrico"
    },
    "routes": [
        {"url": "/sample(.*)?", "controller": "pages/Page.php", "model": "models/sample.json"},
        {"url": "/logout(.*)?", "controller": "pages/Logout.php"},
        {"url": "/login(.*)?", "controller": "pages/Login.php"},
        {"url": "(.*)?", "controller": "pages/Home.php"}
    ],
    "adapters": {
        "MySQL": {
            "host": "localhost",
            "user": "root",
            "pass": "",
            "dbname": "fabrico_test"
        }
    }
}
